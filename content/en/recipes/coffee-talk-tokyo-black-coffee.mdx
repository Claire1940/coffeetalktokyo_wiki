  <Step num="1" title="Select the Coffee Base">

  <Step num="2" title="Identify Customer Preferences">

  <Step num="3" title="Add Modifiers (Optional)">

    - **Sugar**: For sweetness.

  <Step num="4" title="Choose Temperature">

  </Step>
